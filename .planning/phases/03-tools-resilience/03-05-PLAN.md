<objective>
Ship the plain-HTML fallback (OBSV-12) — the safety net that ensures a recruiter who lands when /api/chat is 500ing, or when the classifier is hard-down, never leaves empty-handed. Same URL (`/`), build-time-generated content from KB sources, no agent UI, just bio + roles + a prominent "Email Joe" mailto.

Purpose: The existential failure mode of this project is "recruiter scans QR, lands on broken UI, walks away with worse impression than no agent at all." This plan eliminates that. Two trigger paths: (1) `/api/chat` returns 500 twice → ChatUI client redirects to `/?fallback=1` → page.tsx renders fallback; (2) `/api/health` reports `classifier === 'down'` → page.tsx detects on SC render → renders fallback directly. The third belt-and-suspenders: any uncaught render-time exception in page.tsx → error.tsx renders the same fallback content.

Output: A build-time content extractor script, a gitignored generated constants module, a static fallback component, an extended page.tsx with branched rendering, an error.tsx boundary, and a ChatUI extension for the persistent-500 redirect path.

CONTEXT lock observance:
- D-G-01: minimal content only (bio + last 3 roles + LinkedIn/GitHub/resume PDF + Email Joe CTA). NO case studies, NO stances.
- D-G-02: branched render in page.tsx (research closed: error.tsx is the wrong surface for OBSV-12 because /api/chat 500 is a runtime error, not a render error).
- D-G-03: build-time generation, NEVER request-time read of kb files in the fallback path (cascading-failure avoidance — if Supabase is down and the fallback path read kb live, a Supabase outage would also break the fallback).
- D-G-04: only TWO triggers — chat API persistent 500s + classifier hard-down. Other dep failures degrade in-place via in-character tool fallback (Plan 03-01) or banner deflection (Plan 03-04).

    // scripts/generate-fallback.ts
    // BUILD-TIME ONLY. Reads kb/{about_me,profile,resume}.md → writes
    // src/generated/fallback.ts with static constants the plain-HTML fallback page
    // imports. D-G-03: NEVER read at request time; build-only is the discipline lock
    // (avoids cascading failure where a Supabase outage breaks the fallback).
    //
    // Run by `npm run prebuild` (added to package.json scripts).
    import { readFileSync, writeFileSync, mkdirSync, existsSync } from 'node:fs';
    import path from 'node:path';
    import matter from 'gray-matter';
    import yaml from 'js-yaml';

    const REPO_ROOT = process.cwd();
    const KB = path.join(REPO_ROOT, 'kb');
    const OUT_DIR = path.join(REPO_ROOT, 'src/generated');
    const OUT_FILE = path.join(OUT_DIR, 'fallback.ts');

    function fail(msg: string): never {
      console.error(`generate-fallback: ${msg}`);
      process.exit(1);
    }

    function readSafe(p: string): string {
      if (!existsSync(p)) fail(`missing ${path.relative(REPO_ROOT, p)}`);
      return readFileSync(p, 'utf-8').replace(/\r\n/g, '\n');
    }

    type ProfileShape = {
      links?: { linkedin?: string; github?: string };
      contact?: { email_for_recruiters?: string };
    };

    type Role = { title: string; company: string; dates: string };

    function extractFirstParagraph(content: string, minLen = 100): string {
      // Skip frontmatter (already stripped by gray-matter), find first paragraph >minLen chars
      const paras = content.split(/\n\s*\n/).map((p) => p.trim()).filter(Boolean);
      const first = paras.find((p) => p.length >= minLen);
      return first ?? paras[0] ?? '';
    }

    function extractLastNRoles(resumeMd: string, n: number): Role[] {
      // Heuristic: roles are H2 or H3 headings followed by lines containing "—" or "|"
      // for company + dates. Adapt the regex to the actual resume.md format on first run.
      // Fallback: parse all H2/H3 with the next non-empty line as company/dates.
      const lines = resumeMd.split('\n');
      const roles: Role[] = [];
      for (let i = 0; i < lines.length; i++) {
        const m = lines[i].match(/^#{2,3}\s+(.+)/);
        if (m) {
          const title = m[1].trim();
          // Look for company/dates in next 1-3 lines
          const next = lines
            .slice(i + 1, i + 4)
            .find((l) => /[—–|·]/.test(l) || /\d{4}/.test(l)) ?? '';
          // Crude split — adapt based on actual resume.md format.
          const parts = next.split(/[—–|·]/).map((s) => s.trim()).filter(Boolean);
          const company = parts[0] ?? '';
          const dates = parts[1] ?? next.match(/\d{4}.*\d{4}|\d{4}.*present/i)?.[0] ?? '';
          roles.push({ title, company, dates });
        }
      }
      // Last n (most recent — assuming reverse-chronological order in resume.md).
      // If the resume is chronological-ascending, change to roles.slice(-n).reverse() etc.
      return roles.slice(0, n);
    }

    function main() {
      const aboutRaw = readSafe(path.join(KB, 'about_me.md'));
      const aboutContent = matter(aboutRaw).content.trim();
      const bio = extractFirstParagraph(aboutContent);

      const profileRaw = readSafe(path.join(KB, 'profile.yml'));
      const profile = (yaml.load(profileRaw) ?? {}) as ProfileShape;
      const linkedin = profile.links?.linkedin ?? '';
      const github = profile.links?.github ?? '';
      const email = profile.contact?.email_for_recruiters ?? 'joe.dollinger@gmail.com';

      const resumeRaw = readSafe(path.join(KB, 'resume.md'));
      const resumeContent = matter(resumeRaw).content;
      const roles = extractLastNRoles(resumeContent, 3);

      if (!existsSync(OUT_DIR)) mkdirSync(OUT_DIR, { recursive: true });

      const generated = `// AUTO-GENERATED at build time by scripts/generate-fallback.ts. Do not edit.
    // Regenerate via: npm run build (prebuild hook) or: tsx scripts/generate-fallback.ts
    // Regenerates from kb/about_me.md, kb/profile.yml, kb/resume.md.

    export const FALLBACK_BIO = ${JSON.stringify(bio)};
    export const FALLBACK_LINKEDIN = ${JSON.stringify(linkedin)};
    export const FALLBACK_GITHUB = ${JSON.stringify(github)};
    export const FALLBACK_EMAIL = ${JSON.stringify(email)};
    export const FALLBACK_ROLES: ReadonlyArray<{ title: string; company: string; dates: string }> = ${JSON.stringify(roles, null, 2)};
    `;

      writeFileSync(OUT_FILE, generated, 'utf-8');
      console.log(`generate-fallback: wrote ${path.relative(REPO_ROOT, OUT_FILE)}`);
    }

    main();
    ```

    // src/lib/fetch-health.ts
    // Shared SC helper for fetching /api/health with the 30s revalidate pattern.
    // Used by both src/components/StatusBanner.tsx (Plan 03-04) and
    // src/app/page.tsx (Plan 03-05 branched-render fallback trigger).
    import { headers } from 'next/headers';
    import type { DepStatus } from './health';

    export type HealthShape = {
      anthropic: DepStatus;
      classifier: DepStatus;
      supabase: DepStatus;
      upstash: DepStatus;
      exa: DepStatus;
    };

    export async function fetchHealth(): Promise<HealthShape | null> {
      const h = await headers();
      const host = h.get('x-forwarded-host') ?? h.get('host') ?? 'localhost:3000';
      const proto = h.get('x-forwarded-proto') ?? 'http';
      try {
        const res = await fetch(`${proto}://${host}/api/health`, {
          next: { revalidate: 30 },
        });
        if (!res.ok) return null;
        return (await res.json()) as HealthShape;
      } catch {
        return null;
      }
    }
    ```
