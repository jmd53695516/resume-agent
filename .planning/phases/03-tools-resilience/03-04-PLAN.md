Key strategic decisions (closing CONTEXT Claude's-discretion items):
- **Anthropic-check strategy:** heartbeat-trust (RESEARCH §7) — cost: $0/banner-render; latency: 10-30ms one Redis GET; lag: <60s. Live ping was the alternative; rejected on cost grounds (~$0.03/day if banner renders frequently).
- **Heartbeat key names:** `heartbeat:anthropic` and `heartbeat:classifier` (short form). Coordinated with Plan 03-02 — both plans MUST use identical strings. TTL = 120s (Plan 03-02 writes with `ex: 120`).
- **Heartbeat freshness windows:** 'ok' if <60s old, 'degraded' if 60-120s OR absent (TTL expired). The classic 'down' classification (>300s old) is no longer reachable because TTL=120s causes the key to be absent before 300s — absent → 'degraded' (not 'down') per the heartbeat-trust strategy. A timeout on the Redis GET itself returns 'down' (network failure to Upstash, distinct from heartbeat freshness).
- **Dual-page mount:** new app/chat/layout.tsx (Server Component); /chat/page.tsx stays a Client Component; banner sits as a sibling above the chat client.
- **HTTP 200 always:** no 5xx response from /api/health (D-J-01) — banner consumer handles all-green vs some-degraded entirely render-side.

    // src/lib/health.ts
    // OBSV-07 / OBSV-10. 5 dependency ping helpers consumed by /api/health.
    // Anthropic + classifier use heartbeat-trust (read Redis keys written by
    // /api/chat onFinish per Plan 03-02). Supabase + Upstash + Exa use live
    // pings with a TIMEOUT_MS guard. HTTP 200 always; never throws.
    //
    // Heartbeat key names — MUST match Plan 03-02 writes exactly:
    //   heartbeat:anthropic    (ex: 120)
    //   heartbeat:classifier   (ex: 120)
    //
    // Heartbeat freshness windows (W4-aligned with TTL=120s):
    //   <60s   → 'ok'
    //   60-120s → 'degraded'
    //   absent (TTL expired) → 'degraded'
    //   redis.get throws/timeout → 'down' (network failure, distinct from freshness)
    import { supabaseAdmin } from './supabase-server';
    import { redis } from './redis';

    export type DepStatus = 'ok' | 'degraded' | 'down';

    const TIMEOUT_MS = 1500; // per ping
    const HEARTBEAT_OK_S = 60;
    const HEARTBEAT_DEGRADED_S = 120; // TTL of the heartbeat keys (Plan 03-02)

    async function withTimeout<T>(p: Promise<T>, ms = TIMEOUT_MS): Promise<T> {
      return Promise.race([
        p,
        new Promise<T>((_, rej) => setTimeout(() => rej(new Error('timeout')), ms)),
      ]);
    }

    function classifyHeartbeat(lastOk: string | number | null): DepStatus {
      // Absent (TTL expired or never written) → 'degraded'.
      if (lastOk == null) return 'degraded';
      const ageSec = (Date.now() - Number(lastOk)) / 1000;
      if (ageSec < HEARTBEAT_OK_S) return 'ok';
      // 60s..120s → degraded. After TTL=120s the key is absent (case above).
      return 'degraded';
    }

    export async function pingAnthropic(): Promise<DepStatus> {
      try {
        const lastOk = await withTimeout(
          redis.get<string | number | null>('heartbeat:anthropic'),
        );
        return classifyHeartbeat(lastOk);
      } catch {
        // redis.get itself failed (network to Upstash) — distinguish from "key absent".
        return 'down';
      }
    }

    export async function pingClassifier(): Promise<DepStatus> {
      try {
        const lastOk = await withTimeout(
          redis.get<string | number | null>('heartbeat:classifier'),
        );
        return classifyHeartbeat(lastOk);
      } catch {
        return 'down';
      }
    }

    export async function pingSupabase(): Promise<DepStatus> {
      // W6 fix: explicit .then() chain so the awaitable is unambiguously a Promise.
      // Without .then(), the supabase-js builder is a thenable that can hang in mocks
      // when the test setup forgets to resolve the deepest level.
      try {
        const result = await withTimeout(
          supabaseAdmin
            .from('sessions')
            .select('id', { count: 'estimated', head: true })
            .limit(1)
            .then((r) => r),  // ← W6: materialize the chain to a real Promise
        );
        if (result && typeof result === 'object' && 'error' in result && result.error) {
          return 'degraded';
        }
        return 'ok';
      } catch {
        return 'down';
      }
    }

    export async function pingUpstash(): Promise<DepStatus> {
      try {
        const pong = await withTimeout(redis.ping());
        return pong === 'PONG' ? 'ok' : 'degraded';
      } catch {
        return 'down';
      }
    }

    export async function pingExa(): Promise<DepStatus> {
      try {
        const res = await withTimeout(
          fetch('https://api.exa.ai/', { method: 'HEAD' }),
        );
        return res.ok ? 'ok' : 'degraded';
      } catch {
        return 'down';
      }
    }
