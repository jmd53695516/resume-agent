    ```typescript
    'use client';

    import { useEffect, useState } from 'react';
    import { useRouter } from 'next/navigation';
    import { ChatUI } from '@/components/ChatUI';
    import { ViewToggle } from '@/components/ViewToggle';

    /**
     * Phase 05.2 (D-A-04, D-B-01..03): view-toggle state lives at this page
     * level so the ViewToggle pill (sibling of ChatUI) and ChatUI can both
     * read it. Initial state is the LITERAL 'chat' — no localStorage / no
     * sessionStorage / no prefers-color-scheme (D-B-03). Resets on reload.
     *
     * Body-class side effect adds 'matrix-mode' to <body> when view==='matrix'.
     * Cleanup function strips it on unmount (route change to /admin/* etc.) so
     * the class doesn't leak across navigations (Pitfall 2 in RESEARCH).
     *
     * D-B-02: pill labels are "Light Mode" / "Dark Mode" — the cheeky
     * double-meaning IS the design payoff. DO NOT flatten.
     */
    export default function ChatPage() {
      const router = useRouter();
      const [sessionId, setSessionId] = useState<string | null>(null);
      const [hydrated, setHydrated] = useState(false);
      const [view, setView] = useState<'chat' | 'matrix'>('chat');

      useEffect(() => {
        setHydrated(true);
        const id = typeof window !== 'undefined' ? sessionStorage.getItem('session_id') : null;
        if (!id) {
          router.replace('/');
          return;
        }
        setSessionId(id);
      }, [router]);

      // D-B-01 + Pitfall 2: toggle body class with cleanup. The cleanup is
      // CRITICAL — without it, navigating to /admin/* leaves body.matrix-mode
      // set and the admin page paints itself green.
      useEffect(() => {
        if (view === 'matrix') {
          document.body.classList.add('matrix-mode');
          return () => document.body.classList.remove('matrix-mode');
        }
      }, [view]);

      if (!hydrated || !sessionId) {
        // Brief flash before hydration / redirect. Avoids hydration mismatch
        // (Pitfall 3 — initial 'chat' literal state matches server render).
        return null;
      }

      return (
        <>
          <ViewToggle view={view} onChange={setView} />
          <ChatUI sessionId={sessionId} />
        </>
      );
    }
    ```

    Key decisions per D-B-03 + Pitfall 3:
    - Initial state `useState<'chat' | 'matrix'>('chat')` — NOT `useState(() => sessionStorage.getItem(...))`. Initial render is always 'chat' so SSR + client agree.
    - No `localStorage.getItem('view')`, no `prefers-color-scheme` — explicitly excluded by D-B-03.
    - `<ViewToggle>` is a SIBLING of `<ChatUI>`, not a child. The pill is `position: fixed top-[18px] right-[18px] z-50` so it's not in ChatUI's flex layout.
    - Pitfall 2 cleanup: the inner conditional `if (view === 'matrix')` ensures we only register cleanup when we ADDED the class. Returning a cleanup unconditionally would attempt to remove a class that wasn't there — harmless but wasteful.
    - Per UI-SPEC line 612: ViewToggle renders only AFTER `sessionId` gate passes. The `if (!hydrated || !sessionId) return null` early return covers this.

    Do NOT pass a `view` prop to ChatUI in this plan. Plan 05 will add that when it wires MatrixRain. ChatUI is currently unaware of view; the body-class CSS cascade is sufficient for everything except the canvas mount (Plan 05's responsibility).
