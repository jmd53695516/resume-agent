    ```typescript
    // src/lib/eval/cats/cat4-judge.ts
    // Cat 4 LLM-judge half (EVAL-06). 5 prompts × 5-dim Likert; pass = avg ≥ 4.0
    // both per-case AND aggregate across the 5 cases.
    //
    // Plan 05-07 implements the blind-A/B half (separate runner, not part of this CLI).
    import path from 'node:path';
    import { readFile } from 'node:fs/promises';
    import { childLogger } from '@/lib/logger';
    import { loadCases } from '@/lib/eval/yaml-loader';
    import { writeCase } from '@/lib/eval/storage';
    import { judgeVoiceFidelity } from '@/lib/eval/judge';
    import { callAgent } from '@/lib/eval/agent-client';
    import type { CategoryResult, EvalCase, EvalCaseResult } from '@/lib/eval/types';

    const log = childLogger({ event: 'eval_cat4_judge' });
    const PASS_THRESHOLD = 4.0;
    const MAX_VOICE_SAMPLES = 8;

    async function loadVoiceSamples(): Promise<string[]> {
      const filepath = path.join(process.cwd(), 'kb', 'voice.md');
      const content = await readFile(filepath, 'utf8');
      // Extract paragraphs separated by blank lines; filter to those that look like voice samples
      // (≥ 2 sentences, < 600 chars). Simple heuristic; refine if voice.md uses headers.
      const blocks = content
        .split(/\n\s*\n/)
        .map((b) => b.trim())
        .filter((b) => b.length > 60 && b.length < 600 && !b.startsWith('#') && !b.startsWith('---'));
      return blocks.slice(0, MAX_VOICE_SAMPLES);
    }

    // callAgent imported from src/lib/eval/agent-client.ts (Plan 05-03 Task 4) — DRY across cat runners.

    export async function runCat4Judge(targetUrl: string, runId: string): Promise<CategoryResult> {
      const cases: EvalCase[] = await loadCases(path.join(process.cwd(), 'evals', 'cat-04-prompts.yaml'));
      const voiceSamples = await loadVoiceSamples();
      log.info({ runId, caseCount: cases.length, voiceSampleCount: voiceSamples.length }, 'cat4_judge_started');

      const results: EvalCaseResult[] = [];
      let totalCost = 0;
      let aggregateAvg = 0;

      for (const c of cases) {
        try {
          const { response } = await callAgent({ targetUrl, prompt: c.prompt, sessionId: `eval-cli-cat4-judge-${c.case_id}` });
          const judge = await judgeVoiceFidelity({ response, voiceSamples, caseId: c.case_id });
          const passed = judge.verdict.average >= PASS_THRESHOLD;
          totalCost += judge.cost_cents;
          aggregateAvg += judge.verdict.average;

          const result: EvalCaseResult = {
            case_id: c.case_id,
            category: 'cat4-judge',
            prompt: c.prompt,
            response,
            judge_score: judge.verdict.average,
            judge_verdict: passed ? 'pass' : 'fail',
            judge_rationale: JSON.stringify(judge.verdict),
            passed,
            cost_cents: judge.cost_cents,
          };
          await writeCase({ runId, result });
          results.push(result);
        } catch (e) {
          const result: EvalCaseResult = {
            case_id: c.case_id,
            category: 'cat4-judge',
            prompt: c.prompt,
            response: null,
            judge_score: null,
            judge_verdict: null,
            judge_rationale: `error: ${(e as Error).message}`,
            passed: false,
            cost_cents: 0,
          };
          await writeCase({ runId, result });
          results.push(result);
          log.error({ runId, caseId: c.case_id, err: (e as Error).message }, 'cat4_judge_case_error');
        }
      }

      const aggregate = results.length > 0 ? aggregateAvg / results.length : 0;
      // Pass = ALL per-case ≥ 4.0 AND aggregate ≥ 4.0 (per voice rubric YAML pass_threshold)
      const passed = results.every((r) => r.passed) && aggregate >= PASS_THRESHOLD;

      log.info(
        {
          runId,
          total: results.length,
          per_case_passed: results.filter((r) => r.passed).length,
          aggregate_avg: aggregate.toFixed(2),
          category_passed: passed,
        },
        'cat4_judge_complete',
      );
      return { category: 'cat4-judge', cases: results, passed, cost_cents: totalCost };
    }
    ```
