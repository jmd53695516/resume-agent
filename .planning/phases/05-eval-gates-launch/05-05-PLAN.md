    ```typescript
    // src/lib/eval/cats/cat2.ts
    // Cat 2: tool-use correctness (EVAL-03) — 9 cases.
    // Mostly assertion-based (did the right tool fire? did response shape match?).
    // Includes synthetic spend-cap test (EVAL-10) which mocks the Redis spend
    // counter past threshold and asserts /api/chat deflects gracefully.
    import path from 'node:path';
    import { childLogger } from '@/lib/logger';
    import { redis } from '@/lib/redis';
    import { loadCases } from '@/lib/eval/yaml-loader';
    import { writeCase } from '@/lib/eval/storage';
    import { callAgent as callAgentText } from '@/lib/eval/agent-client';
    import type { CategoryResult, EvalCase, EvalCaseResult } from '@/lib/eval/types';

    const log = childLogger({ event: 'eval_cat2' });

    interface ToolCallEvent { name: string; args: unknown; }
    interface ChatStreamResult { responseText: string; toolCalls: ToolCallEvent[]; httpStatus: number; }

    /**
     * Cat 2 needs BOTH the text channel AND the tool-call events. The shared
     * src/lib/eval/agent-client.ts (Plan 05-03 Task 4) handles the text channel;
     * this helper extends it by parsing tool-call lines from the same rawBody.
     * The SSE prefix codes are documented at the top of agent-client.ts
     * against the captured .eval-tmp/sample-stream.txt fixture.
     */
    async function callAgentWithTools(targetUrl: string, prompt: string, sessionId: string): Promise<ChatStreamResult> {
      const out = await callAgentText({ targetUrl, prompt, sessionId });
      const toolCalls: ToolCallEvent[] = [];
      // Parse tool-call lines from rawBody. Use codes documented in agent-client.ts.
      // EXECUTOR: confirm the codes ('9', '2', 'a') match what's in the comment block
      // at the top of agent-client.ts (which was calibrated against .eval-tmp/sample-stream.txt).
      for (const code of ['9', '2', 'a']) {
        const re = new RegExp(`^${code}:(\\{.*\\})$`, 'gm');
        for (const m of out.rawBody.matchAll(re)) {
          try {
            const obj = JSON.parse(m[1]);
            if (obj.toolName) toolCalls.push({ name: obj.toolName, args: obj.args });
          } catch {
            /* skip malformed */
          }
        }
      }
      return { responseText: out.response, toolCalls, httpStatus: out.httpStatus };
    }

    /** Assertion functions per tool. Return { passed, rationale }. */
    function assertResearch(prompt: string, result: ChatStreamResult): { passed: boolean; rationale: string } {
      const fired = result.toolCalls.some((t) => t.name === 'research_company');
      const paragraphs = result.responseText.split(/\n\n+/).filter((p) => p.trim().length > 30);
      const hasUrls = /https?:\/\/[^\s)]+/.test(result.responseText);
      const passed = fired && paragraphs.length >= 3 && hasUrls;
      return {
        passed,
        rationale: JSON.stringify({ fired, paragraphCount: paragraphs.length, hasUrls }),
      };
    }

    function assertCaseStudy(prompt: string, result: ChatStreamResult): { passed: boolean; rationale: string } {
      const fired = result.toolCalls.some((t) => t.name === 'get_case_study');
      const wordCount = result.responseText.trim().split(/\s+/).length;
      const hasClosingLine = /Want to go deeper, or hear a different story\?/.test(result.responseText);
      const inWordRange = wordCount >= 250 && wordCount <= 600;
      // Edge case: unknown slug → tool fires but no narration; menu returned instead
      const isMenuResponse = /case study|case-study|here are/i.test(result.responseText) && wordCount < 200;
      const passed = fired && (isMenuResponse || (inWordRange && hasClosingLine));
      return {
        passed,
        rationale: JSON.stringify({ fired, wordCount, hasClosingLine, isMenuResponse }),
      };
    }

    function assertMetric(prompt: string, result: ChatStreamResult): { passed: boolean; rationale: string } {
      const fired = result.toolCalls.some((t) => t.name === 'design_metric_framework');
      const tc = result.toolCalls.find((t) => t.name === 'design_metric_framework');
      const args = tc?.args as { description?: string } | undefined;
      // The MetricCard JSON shape is enforced server-side; here we just verify the
      // tool fired and the response includes the 6 section labels somewhere
      const sections = ['north_star', 'input_metrics', 'counter_metrics', 'guardrails', 'experiment', 'open_questions'];
      const responseHasShape =
        sections.filter((s) => result.responseText.toLowerCase().replace(/[ -]/g, '_').includes(s.replace(/[ -]/g, '_'))).length >= 4;
      const passed = fired && responseHasShape;
      return {
        passed,
        rationale: JSON.stringify({ fired, sectionsFound: sections.filter((s) => result.responseText.toLowerCase().replace(/[ -]/g, '_').includes(s.replace(/[ -]/g, '_'))) }),
      };
    }

    function assertSpendCapDeflection(result: ChatStreamResult): { passed: boolean; rationale: string } {
      // Deflection text varies; check for common deflection markers
      const isDeflection = /come back|spend cap|busy day|few hours|rate limit|capacity/i.test(result.responseText);
      const noToolFired = result.toolCalls.length === 0;
      const passed = isDeflection && noToolFired && result.httpStatus === 200;
      return {
        passed,
        rationale: JSON.stringify({ isDeflection, noToolFired, httpStatus: result.httpStatus, snippet: result.responseText.slice(0, 200) }),
      };
    }

    export async function runCat2(targetUrl: string, runId: string): Promise<CategoryResult> {
      const yamlPath = path.join(process.cwd(), 'evals', 'cat-02-tools.yaml');
      const cases: EvalCase[] = await loadCases(yamlPath);
      log.info({ runId, caseCount: cases.length }, 'cat2_started');

      const results: EvalCaseResult[] = [];
      let totalCost = 0;
      const today = new Date().toISOString().slice(0, 10);
      const spendKey = `resume-agent:spend:${today}`;

      for (const c of cases) {
        const sessionId = `eval-cli-cat2-${c.case_id}`;
        let assertion: { passed: boolean; rationale: string };
        let response: string | null = null;
        const isSpendCapCase = c.tags?.includes('spend-cap') ?? false;
        const originalSpend = isSpendCapCase ? await redis.get<number>(spendKey) : null;

        try {
          if (isSpendCapCase) {
            // EVAL-10 synthetic — set spend past threshold (300 cents)
            await redis.set(spendKey, 350);
            log.info({ runId, caseId: c.case_id, spendKey }, 'cat2_spendcap_synthetic_set');
          }

          const out = await callAgentWithTools(targetUrl, c.prompt, sessionId);
          response = out.responseText;

          if (isSpendCapCase) {
            assertion = assertSpendCapDeflection(out);
          } else if (c.tool_expected === 'research_company') {
            assertion = assertResearch(c.prompt, out);
          } else if (c.tool_expected === 'get_case_study') {
            assertion = assertCaseStudy(c.prompt, out);
          } else if (c.tool_expected === 'design_metric_framework') {
            assertion = assertMetric(c.prompt, out);
          } else {
            assertion = { passed: false, rationale: `unknown tool_expected: ${c.tool_expected}` };
          }
        } catch (e) {
          assertion = { passed: false, rationale: `network error: ${(e as Error).message}` };
        } finally {
          // Reset spend-cap synthetic
          if (isSpendCapCase) {
            if (originalSpend != null) {
              await redis.set(spendKey, originalSpend);
            } else {
              await redis.del(spendKey);
            }
            log.info({ runId, caseId: c.case_id }, 'cat2_spendcap_synthetic_reset');
          }
        }

        const result: EvalCaseResult = {
          case_id: c.case_id,
          category: 'cat2',
          prompt: c.prompt,
          response,
          judge_score: null,
          judge_verdict: assertion.passed ? 'pass' : 'fail',
          judge_rationale: assertion.rationale,
          passed: assertion.passed,
          cost_cents: 0, // assertion-based; agent cost not extracted here
        };
        await writeCase({ runId, result });
        results.push(result);
      }

      const passed = results.every((r) => r.passed);
      log.info({ runId, total: results.length, passed: results.filter((r) => r.passed).length }, 'cat2_complete');
      return { category: 'cat2', cases: results, passed, cost_cents: totalCost };
    }
    ```

    ```typescript
    import path from 'node:path';
    import { childLogger } from '@/lib/logger';
    import { loadCases } from '@/lib/eval/yaml-loader';
    import { writeCase } from '@/lib/eval/storage';
    import { judgePersona } from '@/lib/eval/judge';
    import { callAgent } from '@/lib/eval/agent-client';
    import type { CategoryResult, EvalCase, EvalCaseResult } from '@/lib/eval/types';

    const log = childLogger({ event: 'eval_cat3' });
    // callAgent imported from src/lib/eval/agent-client.ts (Plan 05-03 Task 4) — DRY across cat runners.

    export async function runCat3(targetUrl: string, runId: string): Promise<CategoryResult> {
      const cases: EvalCase[] = await loadCases(path.join(process.cwd(), 'evals', 'cat-03-persona.yaml'));
      log.info({ runId, caseCount: cases.length }, 'cat3_started');

      const results: EvalCaseResult[] = [];
      let totalCost = 0;

      for (const c of cases) {
        try {
          const { response } = await callAgent({ targetUrl, prompt: c.prompt, sessionId: `eval-cli-cat3-${c.case_id}` });
          const judge = await judgePersona({
            prompt: c.prompt,
            response,
            personaCriterion: c.expected_pass_criteria ?? 'Stay in persona; refuse warmly.',
            caseId: c.case_id,
          });
          const passed = judge.verdict.verdict === 'pass' && judge.verdict.score >= 4;
          totalCost += judge.cost_cents;

          const result: EvalCaseResult = {
    ```
