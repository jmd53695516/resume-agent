    ```typescript
    // src/lib/eval/calibration.ts
    // Quadratic weighted Cohen's kappa (RESEARCH §11). Pearson r as secondary.
    // Threshold: kappa &lt; 0.5 = recalibration trigger.

    export function pearsonR(x: number[], y: number[]): number {
      if (x.length !== y.length) throw new Error('pearsonR: array length mismatch');
      if (x.length === 0) throw new Error('pearsonR: empty arrays');
      const n = x.length;
      const meanX = x.reduce((s, v) => s + v, 0) / n;
      const meanY = y.reduce((s, v) => s + v, 0) / n;
      let num = 0, denX = 0, denY = 0;
      for (let i = 0; i &lt; n; i++) {
        const dx = x[i] - meanX, dy = y[i] - meanY;
        num += dx * dy; denX += dx * dx; denY += dy * dy;
      }
      const den = Math.sqrt(denX * denY);
      return den === 0 ? 0 : num / den;
    }

    export function quadraticWeightedKappa(humanScores: number[], judgeScores: number[], N = 5): number {
      if (humanScores.length !== judgeScores.length) throw new Error('kappa: length mismatch');
      if (humanScores.length === 0) throw new Error('kappa: empty');
      // Build observed agreement matrix O[i][j], weights w[i][j], expected E[i][j]
      const O = Array.from({ length: N }, () => Array(N).fill(0));
      const histH = Array(N).fill(0), histJ = Array(N).fill(0);
      for (let k = 0; k &lt; humanScores.length; k++) {
        const h = humanScores[k] - 1, j = judgeScores[k] - 1;
        if (h &lt; 0 || h >= N || j &lt; 0 || j >= N) throw new Error('kappa: score out of [1,N] range');
        O[h][j] += 1; histH[h] += 1; histJ[j] += 1;
      }
      const total = humanScores.length;
      const w = Array.from({ length: N }, (_, i) => Array.from({ length: N }, (_, j) => 1 - ((i - j) / (N - 1)) ** 2));
      let numerator = 0, denominator = 0;
      for (let i = 0; i &lt; N; i++) {
        for (let j = 0; j &lt; N; j++) {
          const E = (histH[i] * histJ[j]) / total;
          numerator += (1 - w[i][j]) * O[i][j];
          denominator += (1 - w[i][j]) * E;
        }
      }
      return denominator === 0 ? 0 : 1 - numerator / denominator;
    }

    export interface CategoryAgreementResult { category: string; n: number; kappa: number; pearson: number; }
    export function groupKappaByCategory(
      pairs: Array<{ category: string; human: number; judge: number }>
    ): CategoryAgreementResult[] {
      const groups = new Map&lt;string, { human: number[]; judge: number[] }&gt;();
      for (const p of pairs) {
        if (!groups.has(p.category)) groups.set(p.category, { human: [], judge: [] });
        const g = groups.get(p.category)!;
        g.human.push(p.human); g.judge.push(p.judge);
      }
      return Array.from(groups.entries()).map(([category, g]) => ({
        category,
        n: g.human.length,
        kappa: g.human.length > 1 ? quadraticWeightedKappa(g.human, g.judge) : 0,
        pearson: g.human.length > 1 ? pearsonR(g.human, g.judge) : 0,
      }));
    }
    ```

    ```typescript
    // src/app/api/admin/evals/calibrate/route.ts
    import { requireAdmin } from '@/lib/admin-auth';
    import { supabaseAdmin } from '@/lib/supabase-server';
    import { quadraticWeightedKappa, pearsonR, groupKappaByCategory } from '@/lib/eval/calibration';
    import { nanoid } from 'nanoid';
    import { z } from 'zod';

    const Body = z.object({
      ratings: z.array(z.object({
        eval_case_id: z.string(),
        human_score: z.number().int().min(1).max(5),
        notes: z.string().optional(),
      })).min(1),
    });

    export async function POST(req: Request) {
      const adminCheck = await requireAdmin();
      if (!adminCheck.ok) return new Response('forbidden', { status: 403 });
      const parsed = Body.safeParse(await req.json());
      if (!parsed.success) return Response.json({ error: parsed.error.message }, { status: 400 });

      // Fetch case rows for the rated cases
      const ids = parsed.data.ratings.map((r) => r.eval_case_id);
      const { data: cases } = await supabaseAdmin.from('eval_cases').select('*').in('id', ids);
      if (!cases) return Response.json({ error: 'cases not found' }, { status: 404 });

      const pairs = parsed.data.ratings.map((r) => {
        const c = cases.find((x) => x.id === r.eval_case_id)!;
        return { eval_case_id: r.eval_case_id, judge: Number(c.judge_score), human: r.human_score, category: c.category, notes: r.notes };
      });

      // Insert eval_calibrations rows
      const rows = pairs.map((p) => ({
        id: nanoid(),
        eval_case_id: p.eval_case_id,
        judge_score: p.judge,
        human_score: p.human,
        delta: p.human - p.judge,
        notes: p.notes ?? null,
      }));
      await supabaseAdmin.from('eval_calibrations').insert(rows);

      const kappa = quadraticWeightedKappa(pairs.map((p) => p.human), pairs.map((p) => p.judge));
      const pearson = pearsonR(pairs.map((p) => p.human), pairs.map((p) => p.judge));
      const meanAbsDelta = pairs.reduce((s, p) => s + Math.abs(p.human - p.judge), 0) / pairs.length;
      const perCategory = groupKappaByCategory(pairs.map((p) => ({ category: p.category, human: p.human, judge: p.judge })));

      return Response.json({
        kappa: Number(kappa.toFixed(2)),
        pearson: Number(pearson.toFixed(2)),
        meanAbsDelta: Number(meanAbsDelta.toFixed(2)),
        perCategory,
        recalibrationTriggered: kappa &lt; 0.5,
      });
    }
    ```
