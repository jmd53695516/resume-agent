    ```typescript
    // src/lib/eval/ab-mapping.ts
    // Cat 4 blind A/B mapping (EVAL-05). RESEARCH §10. Server-side mapping in
    // eval_ab_sessions; cookie key only — never client-visible.
    import { nanoid } from 'nanoid';
    import { supabaseAdmin } from '@/lib/supabase-server';
    import { childLogger } from '@/lib/logger';
    import { createRun, updateRunStatus, writeCase } from '@/lib/eval/storage';
    import { JUDGE_MODEL } from '@/lib/eval-models';

    const log = childLogger({ event: 'eval_ab' });

    export interface Snippet { kind: 'ai' | 'joe'; source_id: string; snippet: string; position: number; }

    function fisherYatesShuffle<T>(items: T[]): T[] {
      const a = items.slice();
      for (let i = a.length - 1; i > 0; i--) {
        const j = Math.floor(Math.random() * (i + 1));
        [a[i], a[j]] = [a[j], a[i]];
      }
      return a;
    }

    export async function createAbSession(args: {
      agentParagraphs: Array<{ source_id: string; snippet: string }>;
      realJoeExcerpts: Array<{ source_id: string; snippet: string }>;
    }): Promise<{ sessionId: string; renderedSnippets: Array<{ position: number; snippet: string }> }> {
      if (args.agentParagraphs.length !== 5 || args.realJoeExcerpts.length !== 5) {
        throw new Error('createAbSession requires exactly 5 agent + 5 real-Joe entries');
      }
      const all: Snippet[] = [
        ...args.agentParagraphs.map((a) => ({ kind: 'ai' as const, source_id: a.source_id, snippet: a.snippet, position: 0 })),
        ...args.realJoeExcerpts.map((j) => ({ kind: 'joe' as const, source_id: j.source_id, snippet: j.snippet, position: 0 })),
      ];
      const shuffled = fisherYatesShuffle(all).map((s, i) => ({ ...s, position: i }));
      const sessionId = nanoid();
      const { error } = await supabaseAdmin.from('eval_ab_sessions').insert({
        id: sessionId,
        shuffled_snippets: shuffled,
      });
      if (error) throw new Error(`createAbSession failed: ${error.message}`);
      log.info({ sessionId }, 'eval_ab_session_created');
      return {
        sessionId,
        renderedSnippets: shuffled.map((s) => ({ position: s.position, snippet: s.snippet })),
      };
    }

    export async function validateAndScoreAbSession(args: {
      sessionId: string;
      identifications: boolean[]; // length 10; identifications[i]=true means tester said snippet[i] is AI
      testerRole: 'pm' | 'non-pm' | 'other';
      targetUrl: string;
    }): Promise<{ pct: number; passed: boolean; runId: string }> {
      if (args.identifications.length !== 10) throw new Error('identifications must have length 10');
      const { data, error } = await supabaseAdmin.from('eval_ab_sessions')
        .select('shuffled_snippets, expires_at, submitted_at').eq('id', args.sessionId).single();
      if (error || !data) throw new Error(`session not found: ${args.sessionId}`);
      if (new Date(data.expires_at) < new Date()) throw new Error('session expired');
      if (data.submitted_at) throw new Error('session already submitted');

      const mapping = data.shuffled_snippets as Snippet[];
      let correct = 0;
      for (let i = 0; i < 10; i++) {
        if (args.identifications[i] === (mapping[i].kind === 'ai')) correct++;
      }
      // identification_pct = correct_AI_picks / 5 — 5 AI snippets, perfect identification = 5/5 = 1.0
      const correctAi = mapping.filter((m, i) => m.kind === 'ai' && args.identifications[i]).length;
      const pct = correctAi / 5;
      const passed = pct < 0.70;

      // Write eval_runs row tagged cat4-blind-ab
      const runId = await createRun({ targetUrl: args.targetUrl, judgeModel: JUDGE_MODEL, gitSha: undefined, scheduled: false });
      const result = {
        case_id: `cat4-blind-ab-${args.sessionId}`,
        category: 'cat4-blind-ab' as const,
        prompt: `(blind A/B; tester_role=${args.testerRole})`,
        response: JSON.stringify({ identification_pct: pct, correct_picks_overall: correct }),
        judge_score: pct, judge_verdict: passed ? 'pass' : 'fail',
        judge_rationale: JSON.stringify({ pct, passed, correctAi, correct, testerRole: args.testerRole }),
        passed, cost_cents: 0,
      };
      await writeCase({ runId, result });
      await updateRunStatus({
        runId,
        summary: { totalCases: 1, passed: passed ? 1 : 0, failed: passed ? 0 : 1, totalCostCents: 0, status: passed ? 'passed' : 'failed' },
      });

      // Mark eval_ab_session as submitted
      await supabaseAdmin.from('eval_ab_sessions').update({
        identifications: args.identifications,
        submitted_at: new Date().toISOString(),
        tester_role: args.testerRole,
        eval_run_id: runId,
      }).eq('id', args.sessionId);

      log.info({ sessionId: args.sessionId, runId, pct, passed, testerRole: args.testerRole }, 'eval_ab_scored');
      return { pct, passed, runId };
    }
    ```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Page render → eval_ab_sessions write | Server-component creates mapping; only Joe-as-admin reaches this code |
| Tester click DOM → submit POST | Client component holds state; mapping NOT in DOM (kind labels stripped) |
| Cookie ra_eval_session → POST handler | HTTP-only cookie carries sessionId; server reads mapping by ID |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
