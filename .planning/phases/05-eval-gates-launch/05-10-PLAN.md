        types:
          - 'vercel.deployment.ready'
          - 'scheduled-eval'
      workflow_dispatch: {}

    jobs:
      eval:
        runs-on: ubuntu-latest
        timeout-minutes: 10
        steps:
          - name: Mark check pending
            uses: vercel/repository-dispatch/actions/status@v1
            if: github.event_name == 'repository_dispatch'

              name: 'eval'
              state: 'pending'

          - uses: actions/checkout@v6
            with:
              ref: ${{ github.event.client_payload.git.sha || github.sha }}

          - uses: actions/setup-node@v5
            with:
              node-version: '22'
              cache: 'npm'

          - run: npm ci

          - run: npx playwright install --with-deps chromium

          - name: Run eval suite
            env:
              EVAL_TARGET_URL: ${{ github.event.client_payload.url || github.event.client_payload.target_url }}
              ANTHROPIC_API_KEY: ${{ secrets.ANTHROPIC_API_KEY }}
              GOOGLE_GENERATIVE_AI_API_KEY: ${{ secrets.GOOGLE_GENERATIVE_AI_API_KEY }}
              SUPABASE_SERVICE_ROLE_KEY: ${{ secrets.SUPABASE_SERVICE_ROLE_KEY }}
              NEXT_PUBLIC_SUPABASE_URL: ${{ secrets.NEXT_PUBLIC_SUPABASE_URL }}
              GIT_SHA: ${{ github.event.client_payload.git.sha || github.sha }}
              EVAL_SCHEDULED: ${{ github.event.action == 'scheduled-eval' && '1' || '0' }}
            run: npm run eval

          - name: Mark check final state
            uses: vercel/repository-dispatch/actions/status@v1
            if: github.event_name == 'repository_dispatch' && always()
            with:
              name: 'eval'
              state: ${{ job.status == 'success' && 'success' || 'failure' }}
