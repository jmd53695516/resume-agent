```typescript
interface TextDeltaEvent { type: 'text-delta'; delta: string; }
function isTextDelta(v: unknown): v is TextDeltaEvent { ... }

export function parseChatStream(rawBody: string): string {
  if (!rawBody) return '';
  const parts: string[] = [];
  for (const line of rawBody.split(/\r?\n/)) {
    const trimmed = line.trimStart();
    if (!trimmed.startsWith('data:')) continue;
    const payload = trimmed.slice(5).trimStart();
    if (payload === '' || payload === '[DONE]') continue;
    try {
      const parsed: unknown = JSON.parse(payload);
      if (isTextDelta(parsed)) parts.push(parsed.delta);
    } catch { /* skip */ }
  }
  return parts.join('');
}
```
